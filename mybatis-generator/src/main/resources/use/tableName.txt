public class CninfTableCreator {

    private static String create_format = "CREATE TABLE `%s` (\n" +
            "`id` bigint(11) unsigned NOT NULL AUTO_INCREMENT, \n" +
            "  %s, \n" +
            "PRIMARY KEY (`id`)" +
            " ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='%s';";

    private static String column_format = "`%s` VARCHAR(20) DEFAULT NULL COMMENT '%s'";

    @Data
    static class Field {
        private String fieldName;

        private String fieldChineseName;

        private String fieldType;

        private String describe;

    }

    private void generate(String table, String tableComment, String jsonArray) {
        List<Field> fields = JSON.parseArray(jsonArray, Field.class);
        List<String> columns = Lists.newArrayList();
        fields.forEach(field -> columns.add(String.format(column_format, field.fieldName, StringUtils.isNotBlank(field.describe) ? field.fieldChineseName + " " + field.describe : field.fieldChineseName)));
        String createSql = String.format(create_format, table, columns.stream().collect(Collectors.joining(",\n")), tableComment);
        Logs.info("create sql:\n{}", createSql);
    }
